import numpy as np
class Perceptron:
    def __init__(self, learning_rate=0.05, epochs=120):
        self.eta = learning_rate
        self.epochs = epochs
        self.weights = np.random.randn(2)
        self.bias = 0.5
 
    def sigmoid(self, x):
        return 1 / (1 + np.exp(-x))
       
    def fit(self, X, y):
        for _ in range(self.epochs):
            for i in range(len(X)):
                linear_output = np.dot(X[i], self.weights) + self.bias
                y_pred = self.sigmoid(linear_output)
 
                error = y[i] - y_pred
                self.weights += self.eta * error * X[i]
                self.bias += self.eta * error
        return self
 
    def predict(self, new_fruit):
        linear_output = np.dot(new_fruit, self.weights) + self.bias
        y_pred = self.sigmoid(linear_output)
        return "Apple" if y_pred >= 0.5 else "Not apple"
 
X = np.array([
    [0, 120/200], [0, 130/200], [1, 140/200],
    [1, 150/200], [2, 160/200], [2, 170/200]
])
 
# 0 = Not apple & 1 = apple
y = np.array([0, 0, 0, 1, 1, 1])
 
model = Perceptron(learning_rate=0.1, epochs=500)
model.fit(X, y)

5.1: ENCODER DECODER (SEQUENCE GENERATOR)

import torch
import torch.nn as nn
import torch.optim as optim

SEQ_LEN = 4
HIDDEN = 16
EPOCHS = 100

Simple layers
encoder = nn.GRU(1, HIDDEN, batch_first=True)
decoder = nn.GRU(1, HIDDEN, batch_first=True)
fc = nn.Linear(HIDDEN, 1)

params = list(encoder.parameters()) + list(decoder.parameters()) + list(fc.parameters())
optimizer = optim.Adam(params, lr=0.01)
loss_fn = nn.MSELoss()

Make tiny dataset
def get_data(batch=8):
    x = torch.rand(batch, SEQ_LEN, 1)
    y = torch.flip(x, [1])  
    return x, y

Forward function (encoder → decoder)
def forward(src):

     Encoder: get final hidden state
    hidden = encoder(src)

    Decoder start with zeros
    inp = torch.zeros(src.size(0), 1, 1)

    outputs = []

Predict step‑by‑step
    for _ in range(SEQ_LEN):
        out, hidden = decoder(inp, hidden)
        out = fc(out)
        outputs.append(out)
        inp = out feed prediction again

    return torch.cat(outputs, dim=1)

Training loop (very small)
for e in range(EPOCHS):
    x, y = get_data()

    pred = forward(x)
    loss = loss_fn(pred, y)

    optimizer.zero_grad()
    loss.backward()
    optimizer.step()

    if (e+1) % 20 == 0:
        print("Epoch", e+1, "Loss", round(loss.item(), 4))

Test on one example
x, y = get_data(1)
p = forward(x)

print("\nInput:     ", x.squeeze().numpy())
print("Target:    ", y.squeeze().numpy())
print("Predicted: ", p.detach().squeeze().numpy())

PRACTICAL 6.1: SELF-ATTENTION

import numpy as np
X = np.array([
    [1, 0, 1, 0],
    [0, 2, 0, 2],
    [1, 1, 1, 1]
])
Wq = np.random.rand(4,4)
Wk = np.random.rand(4,4)
Wv = np.random.rand(4,4)
Step 1: Create Query, Key, Value matrices
Q = np.dot(X, Wq)
K = np.dot(X, Wk)
V = np.dot(X, Wv)
Step 2: Calculate Attention Scores
scores = np.dot(Q, K.T)
Step 3: Scale scores
dk = K.shape[1]
scaled_scores = scores / np.sqrt(dk)
Step 4: Softmax function
def softmax(x):
    exp_x = np.exp(x - np.max(x))
    return exp_x / exp_x.sum(axis=-1, keepdims=True)
attention_weights = softmax(scaled_scores)
Step 5: Final output
output = np.dot(attention_weights, V)

print("Query Matrix:\n", Q)
print("Key Matrix:\n", K)
print("Value Matrix:\n", V)
print("Attention Weights:\n", attention_weights)
print("Self Attention Output:\n", output)

PRACTICAL 6.3: MULTI-HEAD ATTENTION 
import numpy as np
Input sequence (3 tokens, embedding dimension = 4)
X = np.array([
    [1,0,1,0],
    [0,2,0,2],
    [1,1,1,1]
])
Parameters
d_model = 4
num_heads = 2
d_k = d_model // num_heads
Weight matrices
WQ = np.random.rand(d_model, d_model)
WK = np.random.rand(d_model, d_model)
WV = np.random.rand(d_model, d_model)
WO = np.random.rand(d_model, d_model)
Step 1: Linear projections
Q = np.dot(X, WQ)
K = np.dot(X, WK)
V = np.dot(X, WV)
Step 2: Split into multiple heads
def split_heads(X, num_heads):
    return X.reshape(X.shape[0], num_heads, d_k)
Q_heads = split_heads(Q, num_heads)
K_heads = split_heads(K, num_heads)
V_heads = split_heads(V, num_heads)
Softmax function
def softmax(x):
    exp_x = np.exp(x - np.max(x))
    return exp_x / exp_x.sum(axis=-1, keepdims=True)

heads_output = []
Step 3: Attention for each head
for i in range(num_heads):
   Qh = Q_heads[:,i,:]
    Kh = K_heads[:,i,:]
    Vh = V_heads[:,i,:]
    scores = np.dot(Qh, Kh.T)
    scaled_scores = scores / np.sqrt(d_k)
    attention_weights = softmax(scaled_scores)
    head_output = np.dot(attention_weights, Vh)
    heads_output.append(head_output)
Step 4: Concatenate heads
concat_heads = np.concatenate(heads_output, axis=1)
Step 5: Final linear projection
output = np.dot(concat_heads, WO)
print("Multi Head Attention Output:\n", output)

PRACTICAL 12: LAYER NORMALIZATION
import torch
# sample input (batch_size, seq_len, d_model)
x = torch.randn(2, 3, 4)
eps = 1e-6
# Step 1: compute mean
mean = x.mean(dim=-1, keepdim=True)
# Step 2: compute variance
var = x.var(dim=-1, keepdim=True, unbiased=False)
# Step 3: normalize
x_norm = (x - mean) / torch.sqrt(var + eps)
# Step 4: scale and shift (learnable params)
gamma = torch.ones(x.size(-1))
beta = torch.zeros(x.size(-1))
output = gamma * x_norm + beta
print(output)
