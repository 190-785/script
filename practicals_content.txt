    [2, 160],
    [2, 170]
])
#Label
y = np.array([0, 0, 0, 1, 1, 1])
 
def step_function(x):
    return 1 if x >= 0 else 0
weights = np.random.randn(2)
bias = 0.5
learning_rate = 0.05
epochs = 120
 
for _ in range(epochs):
    for i in range(len(X)):
        linear_output = np.dot(X[i], weights) + bias
        y_pred = step_function(linear_output)
        error = y[i] - y_pred
        weights += learning_rate * error * X[i]
        bias += learning_rate * error

1.2:Design a ANN base system for the classification of the apple having two feature colors and weight (Binary Classification).(normalization)

import numpy as np
class Perceptron:
    def __init__(self, learning_rate=0.05, epochs=120):
        self.eta = learning_rate
        self.epochs = epochs
        self.weights = np.random.randn(2)
        self.bias = 0.5
 
    def sigmoid(self, x):
        return 1 / (1 + np.exp(-x))
       
    def fit(self, X, y):
        for _ in range(self.epochs):
            for i in range(len(X)):
                linear_output = np.dot(X[i], self.weights) + self.bias
                y_pred = self.sigmoid(linear_output)
 
                error = y[i] - y_pred
                self.weights += self.eta * error * X[i]
                self.bias += self.eta * error
        return self
 
    def predict(self, new_fruit):
        linear_output = np.dot(new_fruit, self.weights) + self.bias
        y_pred = self.sigmoid(linear_output)
        return "Apple" if y_pred >= 0.5 else "Not apple"
 
X = np.array([
    [0, 120/200], [0, 130/200], [1, 140/200],
    [1, 150/200], [2, 160/200], [2, 170/200]
])
 
# 0 = Not apple & 1 = apple
y = np.array([0, 0, 0, 1, 1, 1])
 
model = Perceptron(learning_rate=0.1, epochs=500)
model.fit(X, y)
 
test_fruit = np.array([2, 150/200])
print(f"Prediction: {model.predict(test_fruit)}")

2.3 Predict the next number in a sequence (sequence learning).

import numpy as np
from tensorflow.keras.models import Sequential
from tensorflow.keras.layers import SimpleRNN, Dense
# Input sequences (X) and targets (y)
X = np.array([
    [1, 2, 3],
    [2, 3, 4],
    [3, 4, 5],
    [4, 5, 6]
])

y = np.array([4, 5, 6, 7])

# Reshape for RNN: (samples, timesteps, features)
X = X.reshape((X.shape[0], X.shape[1], 1))

model = Sequential()
model.add(SimpleRNN(10, activation='tanh', input_shape=(3, 1)))
model.add(Dense(1))
model.compile(
    optimizer='adam',
    loss='mean_squared_error'
)
model.fit(X, y, epochs=200, verbose=1)
# Test input
test_input = np.array([[5, 6, 7]])
test_input = test_input.reshape((1, 3, 1))

prediction = model.predict(test_input)
print("Predicted value:", prediction)
